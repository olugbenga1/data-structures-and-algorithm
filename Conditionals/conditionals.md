### If Statement

    {
        if (true) {
            console.log("hello world");
        }
    }

    {
        const value = 2 > 1; // type of value is still a boolean in this case

        if (value) {
            console.log("value is true")
        }
    }

#### If/Else

    {
        const value2 = 1 > 2;
        if (value2) {
            console.log("Hello world");
        } else {
            console.log("Hello people");
        }
    }
